    使用NPM安装依赖：express模块

    npm install express --save
    1

    创建好index.js文件之后，输入以下内容

    var express = require('express');
var app = express();

app.get('/', function(res, rep) {
    rep.send('Hello, word!');
});

app.listen(3000);
